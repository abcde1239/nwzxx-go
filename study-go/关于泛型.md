由于 Go1.18 版本（2022）中更新了泛型,此时大多桑梓相关项目并未使用，因而在此给出早先的使用方法

### interface 模拟

```go
package main

import "fmt"
//使用interface{} 表示任意类型
type Stack struct{
    arr[]interface{}
}
func (s *Stack) Push() (v interface{}){
    s.arr =append(s.arr,v)
}
func (s *Stack) Pop() (v interface{},bool){
    if len(s.arr) == 0 {
        return nil,false
    }
    val := s.arr[len(s.arr)-1]
    s.arr =s.arr[:len(s.arr)-1]
    return val, true
}
func main() {
    s :=Stack{}
    s.Push(10)
    s.Push('abcde')

    v,_ :=s.Pop()
    fmt.Println(v) //abcde

     v,_ :=s.Pop()
    fmt.Println(v)  //10
}


```
